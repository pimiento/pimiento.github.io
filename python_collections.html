import collections

2 Counter

Это подкласс класса dict.

2.1 метод elements

Возвращает итератор над элементами counter-а, где каждый ключ будет повторён столько раз, каково значение этого ключа. 

import collections

c = collections.Counter(a=4, b=2, c=0, d=-2)
print(list(c.elements()))

['a', 'a', 'a', 'a', 'b', 'b']

2.2 метод most_common

Возвращает лист пар в порядке убывания значений. 

import collections
print(collections.Counter('abracadabra').most_common(3))

[('a', 5), ('r', 2), ('b', 2)]

2.3 Сложение и вычитание

import collections

a = collections.Counter(a=4, b=2, c=0, d=-2)
b = collections.Counter(a=1, b=2, c=3, d=4)

print(a + b)
print(a - b)
print(a & b)
print(a | b)
a.subtract(b)
print(a)

3 Deque

import collections

d = collections.deque([1, 2, 3, 4, 5])
d.append(6)
print(d)
d.appendleft(0)
print(d)
print(d.count(2))
d.clear()
print(d)
d.extend(['a', 'b', 'c'])
print(d)
d.extendleft(['x', 'y', 'z'])
print(d)
print(d.pop())
print(d)
print(d.popleft())
print(d)
d.remove('x')
print(d)
d.reverse()
print(d)
d.rotate(2)
print(d)

4 namedtuple

Immutable! 

import collections

Point = collections.namedtuple('Point', ['x', 'y'])
p = Point(x=1, y=2)
print(p)
print(p.x)

Point(x=1, y=2)
1
